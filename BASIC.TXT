import java.util.Scanner;
public class CalculatorApp {
    static int staticCounter = 1;
    int instanceValue = 100;

    public static void main(String[] args) {
        byte byteVar = 10;         // 1 byte – small integer values
        short shortVar = 100;      // 2 bytes – medium range numbers
        int intVar = 1000;         // 4 bytes – default integer type
        long longVar = 10000L;     // 8 bytes – large integer values
        float floatVar = 10.5f;    // 4 bytes – decimal with less precision
        double doubleVar = 20.99;  // 8 bytes – decimal with high precision
        char charVar = 'A';        // 2 bytes – single character
        boolean boolVar = true;    // 1 bit – true or false

        System.out.println("---- Primitive Data Types ----");
        System.out.println("byte: " + byteVar);
        System.out.println("short: " + shortVar);
        System.out.println("int: " + intVar);
        System.out.println("long: " + longVar);
        System.out.println("float: " + floatVar);
        System.out.println("double: " + doubleVar);
        System.out.println("char: " + charVar);
        System.out.println("boolean: " + boolVar);
        Scanner sc = new Scanner(System.in);

        System.out.println("\n---- Calculator Section ----");
        System.out.print("Enter first number: ");
        double num1 = sc.nextDouble(); // double chosen for precision

        System.out.print("Enter second number: ");
        double num2 = sc.nextDouble();
        double sum = num1 + num2;
        double diff = num1 - num2;
        double product = num1 * num2;

        if (num2 != 0) {
            double division = num1 / num2;

            System.out.printf("Addition: %.2f%n", sum);
            System.out.printf("Subtraction: %.2f%n", diff);
            System.out.printf("Multiplication: %.2f%n", product);
            System.out.printf("Division: %.2f%n", division);
        } else {
            System.out.println("Division not possible (cannot divide by zero)");
        }

        double implicitCast = intVar;
        int explicitCast = (int) doubleVar;

        System.out.println("\n---- Type Casting ----");
        System.out.println("Implicit casting (int to double): " + implicitCast);
        System.out.println("Explicit casting (double to int): " + explicitCast);
        CalculatorApp obj = new CalculatorApp();
        obj.showScope();

        System.out.println("\nStatic Variable Value: " + staticCounter);

        sc.close();
    }
    void showScope() {
        int localVar = 50; // Local variable

        System.out.println("\n---- Variable Scope ----");
        System.out.println("Local Variable: " + localVar);
        System.out.println("Instance Variable: " + instanceValue);
    }
}
